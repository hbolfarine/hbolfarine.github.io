    We use the bayesm package

# Hyperparamters for Overfitted Finite Mixture model

# Maximum number of factors
G = 10

# Number of posterior samples
N = 60000

# GMM config - keep all posterior mixtures
Mcmc1 = list(R=N, keep=1,nprint = 0)

# Create data frame for bayesm function
n = length(y.data)
y.mat = matrix(y.data,n,1)
Data1  = list(y = y.mat)

# Generate posterior
Prior1 = list(ncomp = G, Mubar = median(y.data), 
              a = rep(10,G), n = 4, V = matrix(4,1,1))

# Outcome from the Gibbs sampler
out.bayesm = bayesm::rnmixGibbs(Data=Data1, Prior=Prior1, Mcmc=Mcmc1)

# Burn in:
burn_in = 10000

# Thinning:
thin = 50
thinning = seq(burn_in+1,N,by=thin)

# Generate data from model
y.data = data_sim_func("crabs")
kmax = 10

y.data = as.matrix(y.data)
Mubar <- apply(y.data,2,median)

p.dim = dim(y.data)[2]

Vinv <- cov(y.data) %>% solve()
V <- Vinv/p.dim

Prior1 = list(ncomp=kmax, Mubar = Mubar, a = rep(1/200,kmax), A = matrix(1,1,1), nu = p.dim, V = V)

Data1 = list(y = y.data)
R0 <- 60000
Mcmc1 = list(R=R0, keep=1, nprint = 0)
out.bayesm = bayesm::rnmixGibbs(Data=Data1, Prior=Prior1, Mcmc=Mcmc1)

# Removing burn in and thinning
burn_in = 10000
thin = 50
thinning = seq(burn_in+1,R0,by=thin)
R = length(thinning)
